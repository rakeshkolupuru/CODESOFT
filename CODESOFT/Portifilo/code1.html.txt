    <script>
        AOS.init({
            duration: 1500,  
            easing: 'ease-in-out',  
        });

        const textElement = document.getElementById('text');
        const words = ["Welcome to My Portfolio", "I'm Rakesh jack", "Aspired to be a developer."];
        let wordIndex = 0;
        let letterIndex = 0;
        let isTyping = true;

        function typeText() {
            if (isTyping) {
                textElement.textContent += words[wordIndex][letterIndex];
                letterIndex++;

                if (letterIndex === words[wordIndex].length) {
                    isTyping = false;
                    setTimeout(eraseText, 1000);
                } else {
                    setTimeout(typeText, 100); 
                }
            }
        }

        function eraseText() {
            if (!isTyping) {
                textElement.textContent = words[wordIndex].substring(0, letterIndex);
                letterIndex--;

                if (letterIndex === 0) {
                    isTyping = true;
                    textElement.textContent = ""; 
                    wordIndex = (wordIndex + 1) % words.length;
                    setTimeout(typeText, 500); 
                } else {
                    setTimeout(eraseText, 100); 
                }
            }
        }
        typeText();

    </script>
